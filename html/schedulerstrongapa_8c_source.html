68 {

117 static inline Scheduler_Node * _Scheduler_strong_APA_Find_highest_ready(
118  Scheduler_strong_APA_Context *self,
119  uint32_t front,
120  uint32_t rear
121 )
122 {
123  Scheduler_Node *highest_ready;
124  Scheduler_strong_APA_Struct *Struct;
125  const Chain_Node *tail;
126  Chain_Node *next;
127  uint32_t index_assigned_cpu;
128  Scheduler_strong_APA_Node *node;
129  Priority_Control min_priority_num;
130  Priority_Control curr_priority;
131  Per_CPU_Control *assigned_cpu;
132  Scheduler_SMP_Node_state curr_state;
133  Per_CPU_Control *curr_CPU;
134  bool first_task;
135 
136  Struct = self->Struct;
137  //When the first task accessed has nothing to compare its priority against
138  // So, it is the task with the highest priority witnessed so far!
139  first_task = true;
140 
141  while( front <= rear ) {
142  curr_CPU = Struct[ front ].cpu;
143  front = front + 1;
144 
145  tail = _Chain_Immutable_tail( &self->All_nodes );
146  next = _Chain_First( &self->All_nodes );
147 
148  while ( next != tail ) {
149  node = (Scheduler_strong_APA_Node*) STRONG_SCHEDULER_NODE_OF_CHAIN( next );
150  //Check if the curr_CPU is in the affinity set of the node
151  if (
152  _Processor_mask_Is_set(&node->Affinity, _Per_CPU_Get_index(curr_CPU))
153  ) {
154  curr_state = _Scheduler_SMP_Node_state( &node->Base.Base );
155 
156  if ( curr_state == SCHEDULER_SMP_NODE_SCHEDULED ) {
157  assigned_cpu = _Thread_Get_CPU( node->Base.Base.user );
158  index_assigned_cpu = _Per_CPU_Get_index( assigned_cpu );
159 
160  if ( Struct[ index_assigned_cpu ].visited == false ) {
161  rear = rear + 1;
162  Struct[ rear ].cpu = assigned_cpu;
163  Struct[ index_assigned_cpu ].visited = true;
164  // The curr CPU of the queue invoked this node to add its CPU
165  // that it is executing on to the queue. So this node might get
166  // preempted because of the invoker curr_CPU and this curr_CPU
167  // is the CPU that node should preempt in case this node
168  // gets preempted.
169  node->invoker = curr_CPU;
170  }
171  }
172  else if ( curr_state == SCHEDULER_SMP_NODE_READY ) {
173  curr_priority = _Scheduler_Node_get_priority( &node->Base.Base );
174  curr_priority = SCHEDULER_PRIORITY_PURIFY( curr_priority );
175 
176  if ( first_task == true || curr_priority < min_priority_num ) {
177  min_priority_num = curr_priority;
178  highest_ready = &node->Base.Base;
179  first_task = false;
180  //In case this task is directly reachable from thread_CPU
181  node->invoker = curr_CPU;
182  }
183  }
184  }
185  next = _Chain_Next( next );
186  }
187  }
188 
189  return highest_ready;
190 }

192 static inline Scheduler_Node *_Scheduler_strong_APA_Get_highest_ready(
193  Scheduler_Context *context,
194  Scheduler_Node *filter
195 )
196 {
197 
198  //Implement the BFS Algorithm for task departure
199  //to get the highest ready task for a particular CPU
200  //return the highest ready Scheduler_Node and Scheduler_Node filter here points
201  // to the victim node that is blocked resulting which this function is called.
202  Scheduler_strong_APA_Context *self;
203 
204  Per_CPU_Control *filter_cpu;
205  Scheduler_strong_APA_Node *node;
206  Scheduler_Node *highest_ready;
207  Scheduler_Node *curr_node;
208  Scheduler_Node *next_node;
209  Scheduler_strong_APA_Struct *Struct;
210 
211  self=_Scheduler_strong_APA_Get_self( context );
212  //Denotes front and rear of the queue
213  uint32_t front;
214  uint32_t rear;
215 
216  uint32_t cpu_max;
217  uint32_t cpu_index;
218 
219  front = 0;
220  rear = -1;
221 
222  filter_cpu = _Thread_Get_CPU( filter->user );
223  Struct = self->Struct;
224  cpu_max = _SMP_Get_processor_maximum();
225 
226  for ( cpu_index = 0 ; cpu_index < cpu_max ; ++cpu_index ) {
227  Struct[ cpu_index ].visited = false;
228  }
229 
230  rear = rear + 1;
231  Struct[ rear ].cpu = filter_cpu;
232  Struct[ _Per_CPU_Get_index( filter_cpu ) ].visited = true;
233 
234  highest_ready = _Scheduler_strong_APA_Find_highest_ready(
235  self,
236  front,
237  rear
238  );
239 
240  if ( highest_ready != filter ) {
241  //Backtrack on the path from
242  //filter_cpu to highest_ready, shifting along every task.
243 
244  node = _Scheduler_strong_APA_Node_downcast( highest_ready );
245 
246  if( node->invoker != filter_cpu ) {
247  // Highest ready is not just directly reachable from the victim cpu
248  // So there is need of task shifting
249 
250  do {
251  curr_node = &node->Base.Base;
252  next_node = _Thread_Scheduler_get_home_node( node->invoker->heir );
253 
254  _Scheduler_SMP_Preempt(
255  context,
256  curr_node,
257  _Thread_Scheduler_get_home_node( node->invoker->heir ),
258  _Scheduler_strong_APA_Allocate_processor
259  );
260 
261  node = _Scheduler_strong_APA_Node_downcast( next_node );
262  }while( node->invoker != filter_cpu );
263  //To save the last node so that the caller SMP_* function
264  //can do the allocation
265 
266  curr_node = &node->Base.Base;
267  highest_ready = curr_node;
268  }
269  }
270 
271  return highest_ready;
272 }

274 static inline Scheduler_Node *_Scheduler_strong_APA_Get_lowest_scheduled(
275  Scheduler_Context *context,
276  Scheduler_Node *filter_base
277 )
278 {
279  //Checks the lowest scheduled directly reachable task
280 
281  uint32_t cpu_max;
282  uint32_t cpu_index;
283 
284  Thread_Control *curr_thread;
285  Scheduler_Node *curr_node;
286  Scheduler_Node *lowest_scheduled;
287  Priority_Control max_priority_num;
288  Priority_Control curr_priority;
289  Scheduler_strong_APA_Node *filter_strong_node;
290 
291  lowest_scheduled = NULL; //To remove compiler warning.
292  max_priority_num = 0;//Max (Lowest) priority encountered so far.
293  filter_strong_node = _Scheduler_strong_APA_Node_downcast( filter_base );
294 
295  //lowest_scheduled is NULL if affinty of a node is 0
296  _Assert( !_Processor_mask_Zero( &filter_strong_node->Affinity ) );
297  cpu_max = _SMP_Get_processor_maximum();
298 
299  for ( cpu_index = 0 ; cpu_index < cpu_max ; ++cpu_index ) {
300 
301  //Checks if the CPU is in the affinity set of filter_strong_node
302  if ( _Processor_mask_Is_set( &filter_strong_node->Affinity, cpu_index) ) {
303  Per_CPU_Control *cpu = _Per_CPU_Get_by_index( cpu_index );
304 
305  if ( _Per_CPU_Is_processor_online( cpu ) ) {
306  curr_thread = cpu->heir;
307  curr_node = _Thread_Scheduler_get_home_node( curr_thread );
308  curr_priority = _Scheduler_Node_get_priority( curr_node );
309  curr_priority = SCHEDULER_PRIORITY_PURIFY( curr_priority );
310 
311  if ( curr_priority > max_priority_num ) {
312  lowest_scheduled = curr_node;
313  max_priority_num = curr_priority;
314  }
315  }
316  }
317  }
318 
319  return lowest_scheduled;
320 }

406 static inline Scheduler_Node* _Scheduler_strong_APA_Get_lowest_reachable(
407  Scheduler_strong_APA_Context *self,
408  uint32_t front,
409  uint32_t rear,
410  Per_CPU_Control **cpu_to_preempt
411 )
412 {
413  Scheduler_Node *lowest_reachable;
414  Priority_Control max_priority_num;
415  uint32_t cpu_max;
416  uint32_t cpu_index;
417  Thread_Control *curr_thread;
418  Per_CPU_Control *curr_CPU;
419  Priority_Control curr_priority;
420  Scheduler_Node *curr_node;
421  Scheduler_strong_APA_Node *curr_strong_node; //Current Strong_APA_Node
422  Scheduler_strong_APA_Struct *Struct;
423 
424  max_priority_num = 0;//Max (Lowest) priority encountered so far.
425  Struct = self->Struct;
426  cpu_max = _SMP_Get_processor_maximum();
427 
428  while( front <= rear ) {
429  curr_CPU = Struct[ front ].cpu;
430  front = front + 1;
431 
432  curr_thread = curr_CPU->heir;
433  curr_node = _Thread_Scheduler_get_home_node( curr_thread );
434 
435  curr_priority = _Scheduler_Node_get_priority( curr_node );
436  curr_priority = SCHEDULER_PRIORITY_PURIFY( curr_priority );
437 
438  curr_strong_node = _Scheduler_strong_APA_Node_downcast( curr_node );
439 
440  if ( curr_priority > max_priority_num ) {
441  lowest_reachable = curr_node;
442  max_priority_num = curr_priority;
443  *cpu_to_preempt = curr_CPU;
444  }
445 
446  if ( !curr_thread->is_idle ) {
447  for ( cpu_index = 0 ; cpu_index < cpu_max ; ++cpu_index ) {
448  if ( _Processor_mask_Is_set( &curr_strong_node->Affinity, cpu_index ) ) {
449  //Checks if the thread_CPU is in the affinity set of the node
450  Per_CPU_Control *cpu = _Per_CPU_Get_by_index( cpu_index );
451  if ( _Per_CPU_Is_processor_online( cpu ) && Struct[ cpu_index ].visited == false ) {
452  rear = rear + 1;
453  Struct[ rear ].cpu = cpu;
454  Struct[ cpu_index ].visited = true;
455  Struct[ cpu_index ].caller = curr_node;
456  }
457  }
458  }
459  }
460  }
461 
462  return lowest_reachable;
463 }

465 static inline bool _Scheduler_strong_APA_Do_enqueue(
466  Scheduler_Context *context,
467  Scheduler_Node *lowest_reachable,
468  Scheduler_Node *node,
469  Priority_Control insert_priority,
470  Per_CPU_Control *cpu_to_preempt
471 )
472 {
473  bool needs_help;
474  Priority_Control node_priority;
475  Priority_Control lowest_priority;
476  Scheduler_strong_APA_Struct *Struct;
477  Scheduler_Node *curr_node;
478  Scheduler_strong_APA_Node *curr_strong_node; //Current Strong_APA_Node
479  Per_CPU_Control *curr_CPU;
480  Thread_Control *next_thread;
481  Scheduler_strong_APA_Context *self;
482  Scheduler_Node *next_node;
483 
484  self = _Scheduler_strong_APA_Get_self( context );
485  Struct = self->Struct;
486 
487  node_priority = _Scheduler_Node_get_priority( node );
488  node_priority = SCHEDULER_PRIORITY_PURIFY( node_priority );
489 
490  lowest_priority = _Scheduler_Node_get_priority( lowest_reachable );
491  lowest_priority = SCHEDULER_PRIORITY_PURIFY( lowest_priority );
492 
493  if( lowest_priority > node_priority ) {
494  //Backtrack on the path from
495  //_Thread_Get_CPU(lowest_reachable->user) to lowest_reachable, shifting
496  //along every task
497 
498  curr_node = Struct[ _Per_CPU_Get_index(cpu_to_preempt) ].caller;
499  curr_strong_node = _Scheduler_strong_APA_Node_downcast( curr_node );
500  curr_strong_node->invoker = cpu_to_preempt;
501 
502  //Save which cpu to preempt in invoker value of the node
503  while( curr_node != node ) {
504  curr_CPU = _Thread_Get_CPU( curr_node->user );
505  curr_node = Struct[ _Per_CPU_Get_index( curr_CPU ) ].caller;
506  curr_strong_node = _Scheduler_strong_APA_Node_downcast( curr_node );
507  curr_strong_node->invoker = curr_CPU;
508  }
509 
510  //Stop just before the last preemption
511  while( curr_node != Struct[ _Per_CPU_Get_index(cpu_to_preempt) ].caller) {
512  next_thread = curr_strong_node->invoker->heir;
513  next_node = _Thread_Scheduler_get_home_node( next_thread );
514  //curr_node preempts the next_node;
515  _Scheduler_SMP_Preempt(
516  context,
517  curr_node,
518  next_node,
519  _Scheduler_strong_APA_Allocate_processor
520  );
521 
522  curr_node = next_node;
523  curr_strong_node = _Scheduler_strong_APA_Node_downcast( curr_node );
524  }
525 
526  node_priority = _Scheduler_Node_get_priority( curr_node );
527  node_priority = SCHEDULER_PRIORITY_PURIFY( node_priority );
528 
529  _Scheduler_SMP_Enqueue_to_scheduled(
530  context,
531  curr_node,
532  node_priority,
533  lowest_reachable,
534  _Scheduler_SMP_Insert_scheduled,
535  _Scheduler_strong_APA_Move_from_scheduled_to_ready,
536  _Scheduler_strong_APA_Allocate_processor
537  );
538 
539  needs_help = false;
540  } else {
541  needs_help = true;
542  }
543 
544  //Add it to All_nodes chain since it is now either scheduled or just ready.
545  _Scheduler_strong_APA_Insert_ready(context,node,insert_priority);
546 
547  return needs_help;
548 }

550 static inline bool _Scheduler_strong_APA_Enqueue(
551  Scheduler_Context *context,
552  Scheduler_Node *node,
553  Priority_Control insert_priority
554 )
555 {
556  //Idea: BFS Algorithm for task arrival
557  //Enqueue node either in the scheduled chain or in the ready chain
558  //node is the newly arrived node and is not scheduled.
559  Scheduler_strong_APA_Context *self;
560  Scheduler_strong_APA_Struct *Struct;
561  uint32_t cpu_max;
562  uint32_t cpu_index;
563  Per_CPU_Control *cpu_to_preempt;
564  Scheduler_Node *lowest_reachable;
565  Scheduler_strong_APA_Node *strong_node;
566 
567  //Denotes front and rear of the queue
568  uint32_t front;
569  uint32_t rear;
570 
571  front = 0;
572  rear = -1;
573 
574  self = _Scheduler_strong_APA_Get_self( context );
575  Struct = self->Struct;
576 
577  strong_node = _Scheduler_strong_APA_Node_downcast( node );
578  cpu_max = _SMP_Get_processor_maximum();
579 
580  for ( cpu_index = 0 ; cpu_index < cpu_max ; ++cpu_index ) {
581  Struct[ cpu_index ].visited = false;
582 
583  //Checks if the thread_CPU is in the affinity set of the node
584  if ( _Processor_mask_Is_set( &strong_node->Affinity, cpu_index) ) {
585  Per_CPU_Control *cpu = _Per_CPU_Get_by_index( cpu_index );
586 
587  if ( _Per_CPU_Is_processor_online( cpu ) ) {
588  rear = rear + 1;
589  Struct[ rear ].cpu = cpu;
590  Struct[ cpu_index ].visited = true;
591  Struct[ cpu_index ].caller = node;
592  }
593  }
594  }
595 
596  //This assert makes sure that there always exist an element in the
597  // Queue when we start the queue traversal.
598  _Assert( !_Processor_mask_Zero( &strong_node->Affinity ) );
599 
600  lowest_reachable = _Scheduler_strong_APA_Get_lowest_reachable(
601  self,
602  front,
603  rear,
604  &cpu_to_preempt
605  );
606 
607  return _Scheduler_strong_APA_Do_enqueue(
608  context,
609  lowest_reachable,
610  node,
611  insert_priority,
612  cpu_to_preempt
613  );
614 }
